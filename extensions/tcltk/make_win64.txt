all: _tcltk

_tcltk:
	-md $(APLS_RELEASE)\langbindings\tcltk
	devenv tcltk.sln /rebuild "Release|x64"
	@xcopy lib $(APLS_RELEASE)\langbindings\tcltk\lib\ /S /Y
	@copy ..\..\bin\win64\tcl86t.dll $(APLS_RELEASE)\bin\ /Y
	@copy ..\..\bin\win64\tk86t.dll $(APLS_RELEASE)\bin\ /Y

clean:
	devenv tcltk.sln /clean Release
	devenv tcltk.sln /clean Debug
	@-sildel /Q *.ncbp
	@-sildel /Q *.pdb

remove_cvs:
	-rmdir /S /Q CVS
